This is my fabfile which builds and releases a version of my Objective-C project. It updates the version (resetting it to 99.99.99 afterwards), swaps development and live versions of files, builds the system, creates and uploads deltas used for Sparkle and finally uploads the latest version itself. I'm sure it's not perfect but it might help others trying to do the same thing.

{% highlight python %} from fabric.api import local, run, env, settings from fabric.operations import put, prompt import os # Requires py2app import plistlib from plistlib import Plist import zipfile import os,stat, sys from cStringIO import StringIO import urllib2_file import urllib2 import shutil # set env variables hereâ€¦ def ZipDir(inputDir, outputZip): '''Zip up a directory and preserve symlinks and empty directories''' zipOut = zipfile.ZipFile(outputZip, 'w', compression=zipfile.ZIP_DEFLATED) rootLen = len(os.path.dirname(inputDir)) def _ArchiveDirectory(parentDirectory): contents = os.listdir(parentDirectory) #store empty directories if not contents: #http://www.velocityreviews.com/forums/t318840-add-empty-directory-using-zipfile.html archiveRoot = parentDirectory[rootLen:].replace('\\', '/').lstrip('/') zipInfo = zipfile.ZipInfo(archiveRoot+'/') zipOut.writestr(zipInfo, '') for item in contents: fullPath = os.path.join(parentDirectory, item) if os.path.isdir(fullPath) and not os.path.islink(fullPath): _ArchiveDirectory(fullPath) else: archiveRoot = fullPath[rootLen:].replace('\\', '/').lstrip('/') if os.path.islink(fullPath): # http://www.mail-archive.com/python-list@python.org/msg34223.html zipInfo = zipfile.ZipInfo(archiveRoot) zipInfo.create_system = 3 # long type of hex val of '0xA1ED0000L', # say, symlink attr magic... zipInfo.external_attr = 2716663808L zipOut.writestr(zipInfo, os.readlink(fullPath)) else: zipOut.write(fullPath, archiveRoot, zipfile.ZIP_DEFLATED) _ArchiveDirectory(inputDir) zipOut.close() def build(version): """ Build with a specified version """ with settings(warn_only=True): plist = Plist.fromFile('SeaweedPlayer-Info.plist') plist['CFBundleVersion'] = version plist['CFBundleShortVersionString'] = version plistlib.writePlist(plist, 'SeaweedPlayer-Info.plist') os.rename('SeaweedPlayer/Configuration.m', 'SeaweedPlayer/ConfigurationDev.m') os.rename('SeaweedPlayer/ConfigurationLive.m', 'SeaweedPlayer/Configuration.m') result = local('xcodebuild -project SeaweedPlayer.xcodeproj -target Release') plist['CFBundleVersion'] = '99.99.99' plist['CFBundleShortVersionString'] = '99.99.99' plistlib.writePlist(plist, 'SeaweedPlayer-Info.plist') os.rename('SeaweedPlayer/Configuration.m', 'SeaweedPlayer/ConfigurationLive.m') os.rename('SeaweedPlayer/ConfigurationDev.m', 'SeaweedPlayer/Configuration.m') if result.failed: return False return True def release(version, deltas=''): if build(version): ZipDir("build/Release/SeaweedPlayer.app", "seaweed_" + version + ".zip") # At this point we have a seaweed_1.0.zip file containing the .app shutil.copytree("build/Release/SeaweedPlayer.app", "old_builds/" + version + ".app", symlinks=True) # Create deltas from specified versions deltas = deltas.split(':') for delta in deltas: local("./BinaryDelta create old_builds/" + delta + ".app/ old_builds/" + version + ".app/ delta_" + delta + "_" + version) # Upload the file put("delta_" + delta + "_" + version, "/home/public/updates/delta_" + delta + "_" + version) # Sign it signature = local("openssl dgst -sha1 -binary < \"delta_" + delta + "_" + version + "\" | openssl dgst -dss1 -sign \"dsa_priv.pem\" | openssl enc -base64", capture=True) # Insert into the database run('echo "use seaweed; INSERT INTO update_deltas SET to_version = \'%s\', from_version = \'%s\', signature = \'%s\';"|mysql --batch --user=%s --password=%s --host=%s' % (version, delta, signature, env.mysqluser, env.mysqlpassword, env.mysqlhost), pty=True) local("rm delta_" + delta + "_" + version) # Sign it signature = local("openssl dgst -sha1 -binary < \"seaweed_" + version + ".zip\" | openssl dgst -dss1 -sign \"dsa_priv.pem\" | openssl enc -base64", capture=True) if os.path.isfile("description.txt"): description = open('description.txt').read() local("rm description.txt") else: description = prompt('What is the description of the update?') # Upload the file put('seaweed_' + version + '.zip', '/home/public/updates/seaweed_' + version + '.zip') # Insert into the database run('echo "use seaweed; INSERT INTO updates SET version = \'%s\', description = \'%s\', signature = \'%s\';"|mysql --batch --user=%s --password=%s --host=%s' % (version, description, signature, env.mysqluser, env.mysqlpassword, env.mysqlhost), pty=True) # Update front page run("echo ' _version.php") local("rm seaweed_" + version + ".zip;") else: print "Build Error" {% endhighlight %}
